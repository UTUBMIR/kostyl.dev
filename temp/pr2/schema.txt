- last_name VARCHAR(64) NOT NULL

- description (TEXT, NULL)

- description (TEXT, NULL)
- cover_image_path (VARCHAR(2048), NULL)

- id (PK) UUID

- password_hash VARCHAR(128) NOT NULL
- email (VARCHAR(376, NULL)
- avatar_path (VARCHAR(2048), NULL)

collections
- id (PK) UUID
- user_id (FK) UUID NULL
- name (VARCHAR(128, NULL) NOT NULL
- created_at (DATATIME)

- audiobook_id (PK) (FK) UUID

FK(collection_id, audiobook_id)

audiobook_files
- id (PK)
- audiobook_id (FK) NOT NULL
- file_path VARCHAR(2048) NOT NULL
- format VARCHAR(10) NOT NULL
- size (INTEGER)

listening_progress
- id (PK) UUID
- user_id (FK) UUID NULL
- audiobook_id (FK) UUID NOT NULL
- position (INTEGER) NOT NULL
- last_listened (DATATIME)
FK(user_id, audiobook_id)

entity Author {
  Id <<key>>
  Name {
    First Name
    Last Name
  }
  Bio
  Image Path
}

entity Genre {
  Id <<key>>
  Name
  Description
}

entity Audiobook <<weak>> {
  Id <<key>>
  Title
  Duration
  Release Year
  Description
  Cover Image Path
}

relationship Written <<identifying>> {
}

Written =1= Author
Written -N- Audiobook

relationship Describe <<identifying>> {
}

Describe =1= Genre
Describe -N- Audiobook

entity User {
  Id <<key>>
  Username
  Password Hash
  Avatar Path
}

entity Collection <<weak>> {
  Id <<key>>
  Name
  Created At
}

relationship Owns <<identifying>> {
}

Owns =1= User
Owns -N- Collection 

relationship Includes {
}

Includes -N- Audiobook
Includes -N- Collection


entity Audiobook_File {
  Id <<key>>
  File Path
  Format
  Size
}

relationship Consists_Of <<identifying>> {
}

Consists_Of =1= Audiobook
Consists_Of -N- Audiobook_File


relationship Listening_Progress <<identifying>> {
  Id <<key>>
  Position
  Last Listened
}

Listening_Progress -N- User
Listening_Progress -N- Audiobook

@endchen
